systemctl stop postgresql

rm -rf /var/lib/postgresql/12/main/*


su - postgres
pg_basebackup -h <primary-ip> -D /var/lib/postgresql/<version>/main -U replicator_user
pg_basebackup -h <primary-ip> -D /var/lib/postgresql/<version>/main -U replicator_user -Fp -Xs -P


# A server enters standby mode if a standby.signal file exists in the data directory when the server is started.
touch /var/lib/postgresql/12/main/standby.signal







sudo systemctl start postgresql


select pg_is_in_recovery();
